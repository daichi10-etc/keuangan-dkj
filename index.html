function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); let sheet; if (data.tipe === "pemasukan") { sheet = ss.getSheetByName("PEMASUKAN"); sheet.appendRow([ new Date(), data.penginput || "", data.sumber || "", data.jenisPemasukan || "", Number(data.cash || 0), Number(data.online || 0), Number(data.total || 0), data.catatan || "", data.foto || "" ]); } if (data.tipe === "pengeluaran") { sheet = ss.getSheetByName("PENGELUARAN"); sheet.appendRow([ new Date(), data.penginput || "", data.cabang || "", data.kategori || "", data.namaBarang || "", Number(data.cash || 0), Number(data.online || 0), Number(data.total || 0), data.sumberUang || "", data.catatan || "", data.foto || "" ]); } if (data.tipe === "utang") { sheet = ss.getSheetByName("UTANG"); sheet.appendRow([ new Date(), data.namaUtang || "", data.jenisTransaksi || "", Number(data.tambahanUtang || 0), Number(data.pembayaran || 0), Number(data.sisaUtang || 0), data.jatuhTempo || "", data.status || "", data.catatan || "", data.foto || "" ]); } return ContentService .createTextOutput("success") .setMimeType(ContentService.MimeType.TEXT); }
